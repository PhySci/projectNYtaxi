def draw_region(ax, row):
    lons = row[['west', 'west', 'east', 'east']].values
    lats = row[['north', 'south', 'south', 'north']].values
    x, y = m(lons, lats)
    xy = zip(x, y)

    color = cm.YlOrRd(row['rel_count'])
    poly = mplPolygon(xy, facecolor=color, alpha=1 if row['total_rides'] > 0 else 0)
    ax.add_patch(poly)
